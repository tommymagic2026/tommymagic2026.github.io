JavaScript 调用接口

// 音频管理器全局对象
window.audioManager

// 播放背景音乐
audioManager.playBgm('start')   // 播放菜单音乐
audioManager.playBgm(1)       // 播放第一关音乐
audioManager.playBgm('boss')    // 播放Boss音乐

// 停止背景音乐
audioManager.stopBgm()

// 播放音效
audioManager.playSfx('laser')  // 播放射击音效

// 设置音量 (0.0 - 1.0)
audioManager.setBgmVolume(0.6)
audioManager.setSfxVolume(0.4)

// 获取所有音频配置 (供Python游戏调用)
audioManager.getConfig()
